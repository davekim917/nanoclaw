**Interface / signature:**

import { z } from 'zod';

export function registerProviderConfigSchema(name: string, schema: z.ZodType): void;
export function getProviderConfigSchema(name: string): z.ZodType | undefined;

  | { ok: true; data: Record<string, unknown> }
